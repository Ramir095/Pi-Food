/// getAllRecipes en async await

export const getAllRecipes = () => {
    return async function (dispatch) {
        dispatch({type: START_LOADING})
        const all = await axios.get(`${URL_C}/recipes`)
        const response = await all.data
        dispatch({
            type: GET_ALL_RECIPES,
            payload: response
        })  
    }
}
